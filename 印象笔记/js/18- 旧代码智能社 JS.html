function startMove(obj, attr, iTarget) {
  clearInterval(obj.timer);
  obj.timer = setInterval(function() {
    var cur = 0; //1、parseInt(getStyle(obj, attr))，parseInt对透明度无效，因为透明度会从0.3->0。
    if (attr == "opacity") {
      cur = Math.round(parseFloat(getStyle(obj, attr) * 100));
      //Math.round()四舍五入解决计算机计算小数时相似值的问题。
    } else {
      cur = parseInt(getStyle(obj, attr));
    };
    var speed = (iTarget - cur) / 6;
    speed = speed > 0 ? Math.ceil(speed) : Math.floor(speed);
    if (cur == iTarget) {
      clearInterval(obj.timer);
    } else {
      if (attr == "opacity") {
        obj.style.filter = "alpha: opacity(" + cur + ")"
        obj.style[attr] = (cur + speed) / 100;
      } else {
        obj.style[attr] = cur + speed + "px";
      }
    };
  }, 30)
};

function ajax(url, fnSucc, fnFaild)
{
  //1.创建Ajax对象
  if(window.XMLHttpRequest)//用没有定义的变量会报错。用没有定义的属性返回的是undefined
  {
    var oAjax=new XMLHttpRequest();
  }
  else
  {
    var oAjax=new ActiveXObject("Microsoft.XMLHTTP");//ActiveX：不是插件的插件，IE专用，为了兼容IE6
  }
  
  //2.连接服务器（打开和服务器的连接）
  oAjax.open('GET', url, true);//open(方法名,文件,异步传输)
  
  //3.发送
  oAjax.send();
  
  //4.接收返回 浏览器和服务器交互时触发
  oAjax.onreadystatechange=function ()
  {
    if(oAjax.readyState==4)//oAjax.readyState 浏览器和服务器交互到哪一步了 4代表解析响应数据完成
    {
      if(oAjax.status==200)
      {
        //alert('成功了：'+oAjax.responseText);
        fnSucc(oAjax.responseText);
      }
      else
      {
        //alert('失败了');
        if(fnFaild)
        {
          fnFaild();
        }
      }
    }
  };
}
